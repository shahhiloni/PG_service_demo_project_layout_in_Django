id INTEGER PRIMARY KEY AUTOINCREMENT, 
name TEXT / VARCHAR  (String)

)

INSERT INTO students (name, age, email) 
VALUES ("meet", 20, "meet@gmail.com")

# step: 3 : view the data which is already exist in the table 

SELECT name, age, email FROM students
